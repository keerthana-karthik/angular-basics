    Parent to child Data Transfer
        @Input for Binding to Custom Properties
        @Input('aliasName') in ts, [aliasName] in html
        In Parent Component
            <app-child
                *ngFor="let item of array_of_children; let i = index"
                [valueFromParent]="valueFromParent"
                (childValueChanged)="onChildValueUpdated($event)">
            </app-child>
        In Child Componnet
            @Input() valueFromParent: string;

    Child to Parent Data Transfer
        @Output for Binding to Custom Events
        @Output('aliasName') in ts, (aliasName) in html
        In Child Componnet
            @Output() childValueChanged = new EventEmitter<string>()
        In Parent Component
            <app-child
                *ngFor="let item of array_of_children; let i = index"
                [valueFromParent]="valueFromParent"
                (childValueChanged)="onChildValueUpdated($event)">
            </app-child>
